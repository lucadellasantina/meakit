1function [ C1,P1,H1 ] = glia_Ca_3_3(C,P,H,location_release_amount,Row,Col)  %#codegen
 2%GLIA_CA Summary of this function goes here
 3
 4%% Parameters
 5deltaT=0.001;       %% Time step interval (s),i.e. 1ms
 6taoh=2.0;           %% the time constant for the dynamics of h, unit : s
 7b=0.111;            %% Fraction of open IP3 receptors at [Ca2+]=0
 8V1=0.889;           %% the proportion of IP3Rs that are activated by the binding of Ca2+
 9kp=0.17;            %% the linear rate of IP3 breakdown, unit : /s
10mu0=0.567;          %% the proportion of IP3Rs that are activated at [IP3] =0μM
11mu1=0.433;          %% the proportion of IP3Rs that are activated by bound IP3
12
13kflux=16.2*400;     %% the maximum total Ca2+ flux through all IP3Rs, unit : μM/s
14gamma=2.0*400;      %% the maximmum rate of Ca2+ pumping from the cytosol, unit : μM/s
15k1=0.7*400;         %% Kinetic constant, unit : μM
16k2=0.7*400;         %% Kinetic constant, unit : μM
17kgamma=0.1*400;     %% Kinetic constant of calcium pumps of ER, unit : μM
18Dp=300/400;         %% the cytosolic diffusion coefficients of IP3, unit : (μm)^2/s
19Dc=20/400;          %% the cytosolic diffusion coefficients of Ca2+, unit : (μm)^2/s
20kmu=4*400;          %%  Kinetic constant of mu, unit : μM
21peq=0.28*400;       %% IP3 resting concentration, unit : uM
22
23Fp=5/20;            %% IP3 permeability of the cell membrane, unit : μm/s
24Fc=1.1/20;          %% Ca2+ permeability of the cell membrane, unit : μm/s
25
26C1=ones(150,150);   %% the cytosolic free calcium concentration, unit : μM
27P1=ones(150,150);   %% the cytosolic IP3 concentration, unit : μM
28H1=ones(150,150);   %% the proportion of IP3Rs that have not been closed by Ca2+
29
30
31%% Calculation
32ip3_rel=find(location_release_amount(:,3)~=0);
33for i=1:length(ip3_rel)
34    x=ceil(location_release_amount(ip3_rel(i),1)/20);
35    y=ceil(location_release_amount(ip3_rel(i),2)/20);
36    P(x,y)=P(x,y)+location_release_amount(ip3_rel(i),3)*0.009;
37end
38
39C=C*400;
40P=P*400;
41
42for j=2:(Col-1)
43    for i=2:(Row-1)
44        old_cij = C(i,j);
45        old_pij = P(i,j);
46        old_hij = H(i,j);
47        C1(i,j)=old_cij+deltaT*(Dc*(C(i+1,j)-2*old_cij+C(i-1,j)+C(i,j+1)-2*old_cij+C(i,j-1))+kflux*(mu0+mu1*P(i,j)/(kmu+P(i,j)))*H(i,j)*(b+V1*old_cij/(k1+old_cij))-gamma*old_cij/(kgamma+old_cij));
48        P1(i,j)=old_pij+deltaT*(Dp*(P(i+1,j)-2*old_pij+P(i-1,j)+P(i,j+1)-2*old_pij+P(i,j-1))+kp*(peq-old_pij));
49        H1(i,j)=old_hij+deltaT*(k2^2/(k2^2+old_cij^2)-old_hij)/taoh;
50        if(mod(i,3)==0)
51            C1(i,j)=C1(i,j)-deltaT*(old_cij-C(i+1,j))*Fc;
52            C1(i+1,j)=C1(i+1,j)+deltaT*(old_cij-C(i+1,j))*Fc;
53            P1(i,j)=P1(i,j)-deltaT*(old_pij-P(i+1,j))*Fp;
54            P1(i+1,j)=P1(i+1,j)+deltaT*(old_pij-P(i+1,j))*Fp;
55        end
56        if(mod(j,3)==0)
57            C1(i,j)=C1(i,j)-deltaT*(old_cij-C(i,j+1))*Fc;
58            C1(i,j+1)=C1(i,j+1)+deltaT*(old_cij-C(i,j+1))*Fc;
59            P1(i,j)=P1(i,j)-deltaT*(old_pij-P(i,j+1))*Fp;
60            P1(i,j+1)=P1(i,j+1)+deltaT*(old_pij-P(i,j+1))*Fp;
61        end
62    end
63end
64
65C1(1,2:(Col-1))=C1(2,2:(Col-1));
66C1(Row,2:(Col-1))=C1(Row-1,2:(Col-1));
67C1(2:(Row-1),1)=C1(2:(Row-1),2);
68C1(2:(Row-1),Col)=C1(2:(Row-1),(Col-1));
69C1(1,1)=C1(2,2);
70C1(1,Col)=C1(2,(Col-1));
71C1(Row,1)=C1(Row-1,2);
72C1(Row,Col)=C1(Row-1,(Col-1));
73
74P1(1,2:(Col-1))=P1(2,2:(Col-1));
75P1(Row,2:(Col-1))=P1(Row-1,2:(Col-1));
76P1(2:(Row-1),1)=P1(2:(Row-1),2);
77P1(2:(Row-1),Col)=P1(2:(Row-1),(Col-1));
78P1(1,1)=P1(2,2);
79P1(1,Col)=P1(2,(Col-1));
80P1(Row,1)=P1(Row-1,2);
81P1(Row,Col)=P1(Row-1,(Col-1));
82
83H1(1,2:(Col-1))=H1(2,2:(Col-1));
84H1(Row,2:(Col-1))=H1(Row-1,2:(Col-1));
85H1(2:(Row-1),1)=H1(2:(Row-1),2);
86H1(2:(Row-1),Col)=H1(2:(Row-1),(Col-1));
87H1(1,1)=H1(2,2);
88H1(1,Col)=H1(2,(Col-1));
89H1(Row,1)=H1(Row-1,2);
90H1(Row,Col)=H1(Row-1,(Col-1));
91
92C1=C1/400;
93P1=P1/400;
94end
