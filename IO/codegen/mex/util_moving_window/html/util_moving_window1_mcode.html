1function [ result ] = util_moving_window( x, window_length, mode, func, varargin ) %#codegen
 2%UTIL_MOVING_WINDOW Applying a @func handle to window-ed data
 3%   Input:
 4%           x:              1-D data
 5%           window_length:  Window width
 6%           mode:           'Cont' or 'Jump'
 7%           func:           Function handle:  Y=F(X)
 8%           funcargs:       Function args: Y=F(X, ...), if needed.
 9%
10%   Output:
11%           result:         In most cases, as a 1-D vector.
12%
13%   Created on Jun/09/2012 By Pu Jiangbo
14%   Britton Chance Center for Biomedical Photonics
15
16% Init variables
17n_len = length(x);
18if window_length < 1 || window_length > n_len
19    error('Wrong input: window_length > 1, < length of x.')
20end
21w_len = window_length;
22
23if strcmpi(mode, 'cont')
24    mode = 0;
25elseif strcmpi(mode, 'jump')
26    mode = 1;
27else
28    error('Wrong input: mode should be CONT or JUMP.');
29end
30
31if mode == 0
32    % Smoothly sliding window
33    % start_pos -> start_pos + w_len - 1
34    % start_pos ++
35    % START: 1
36    % END: if start_pos + w_len - 1>= n_len => start_pos -> n_len
37    result = zeros(n_len-w_len+1);
38    for i = 1:n_len
39        result(i) = func(x(i:i+w_len-1), varargin);
40        %disp(x(i:i+w_len-1));disp(', ')
41        if i+w_len-1 >= n_len
42            result(i+1) = func(x(i+1:end), varargin);
43            %disp(x(i+1:end));disp(', end')
44            break;
45        end
46    end
47elseif mode == 1
48    % Jumping window
49    % start_pos -> start_pos + w_len - 1
50    % start_pos = start_pos + w_len
51    % START: 1
52    % END: if start_pos + w_len + w_len >= n_len => start_pos + w_len -> n_len
53    %      It means: if the next round of next round cannot complete a window, then
54    %      extend the next round.
55    result = zeros(fix(n_len/w_len), 1);
56    result_i = 1;
57    for i = 1:w_len:n_len
58        result(result_i) = func(x(i:i+w_len-1), varargin);
59        %disp(x(i:i+w_len-1));disp(', ')
60        result_i = result_i + 1;
61        if i+w_len+w_len >= n_len
62            result(result_i) = func(x(i+w_len:end), varargin);
63            %disp(x(i+w_len:end));disp(', end')
64            break;
65        end
66    end
67end
68
69end
