1function result = util_sub_normalized_dot_product(a, b) %#codegen
 2%NORMALIZED_DOT_PRODUCT Calculating the dot product of a,b then divides it
 3%using norm(a) * norm(b)
 4%
 5%   Created on Jul/22/2010 By Pu Jiangbo
 6%   Britton Chance Center for Biomedical Photonics
 7%
 8%   $Revision:
 9%       PJB#2010-08-28  Speeding, use A'*B instead of dot
10%       PJB#2012-07-04  Adding assert for codegen
11
12% Assert Class for Codegen
13assert(isscalar(a));
14assert(isscalar(b));
15
16% This is slow
17% dotproduct = dot(a, b);
18% Because when a and b are both column vectors, a'*b is the same as dot(a,b)
19dotproduct = a'*b;
20
21% Avoid NaN (0 / 0 = NaN)
22if ~dotproduct
23    result = 0;
24else
25    result = dotproduct / (norm(a) * norm(b));
26end
27
28end
