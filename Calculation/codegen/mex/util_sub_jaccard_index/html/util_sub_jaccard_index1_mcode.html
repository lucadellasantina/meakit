1function result = util_sub_jaccard_index(a, b)  %#codegen
 2% JACCARD_INDEX Calculating the Jaccard Coefficient Index between a and b.
 3%
 4%   This is the extented jaccard index, a.k.a. Tanimoto coefficient
 5%
 6%   Created on Aug/28/2010 By Pu Jiangbo
 7%   Britton Chance Center for Biomedical Photonics
 8
 9% This is slow
10% dotproduct = dot(a, b);
11% Because when a and b are both column vectors, a'*b is the same as dot(a,b)
12dotproduct = a'*b;
13
14% Avoid nan (0/0 = NaN)
15if ~dotproduct
16    result = 0;
17else
18    result = dotproduct / (norm(a) ^ 2 +  norm(b) ^ 2 - dotproduct);
19end
20
21end
